function _init()
 palt(0, false)
 palt(14, true)
 menuitem(1,"manual",function() toggle_manual() end)
 state = "start"
end

function toggle_manual()
 if state != "manual" then
  prev_state = state
  state="manual"
  page = 1
 else
  state=prev_state
  prev_state = nil
 end
end

function init_hands()
 hands = {{}}
 for p = 1,2 do
  hands[p] = {}
  for i=1,2 do
   //add empty slots
   add(hands[p],{t=0, s=0})
  end
  add(hands[p],{t=p, s=bee})
  for i=1,3 do
   add(hands[p],{t=p, s=ant})
  end
  for i=1,2 do
   add(hands[p],{t=p, s=spi})
  end
  for i=1,3 do
   add(hands[p],{t=p, s=gra})
  end
  for i=1,2 do
   add(hands[p],{t=p, s=btl})
  end
  for i=1,3 do
   //add empty slots
   add(hands[p],{t=0, s=0})
  end
 end
end

function shuffle_board()
	for row = 2,b-1 do
	 board[row] = {}
	 for col = 1,b do
	  //leave out off screen cells
	  if not (row%2 == 1 and col == b) then
	   board[row][col] = {}
	   local sprite = flr(rnd(20)) + 1
	   local team
	   if sprite > 5 then
	   	sprite = 0
	   	team = 0
	   else
	    team = flr(rnd(2)) + 1
	   end
	   board[row][col] = {t=team, s=sprite}
	  end
	 end
	end
end

function init_board()

end
